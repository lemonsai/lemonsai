    • vue 公众号h5获取code的一些问题

TIP

bug：由于code只能使用一次 在公众号刷新页面时会重复使用同一个code

解决：在首次获取code时用sesstion存一遍 在created判断当前url上的code和sesstion里的code是否一致 如果一致就重新授权

export default {
  components: {},
  data() {
    return {
      code: "", //微信code
      account: "", //账号
      password: "", //密码
      btnTxt: "立即绑定",
      getUserData: "",
      wxToken: "",
    };
  },

  created() {
    let code = sessionStorage.getItem("code");
    //判断url上的code和sesstion里的code是否一致  如果一致就重新授权   
    if (this.getUrlParam("code", window.location.href) == code) {
      this.getUrl()
    } else if (this.code == "") {
      this.getCode();
    }

    
  },
  methods: {
    submit() {
      if (this.account == "") {
        this.$toast("账号不能为空");
      } else if (this.password == "") {
        this.$toast("密码不能为空");
      } else {
        // 判断是否m_id 有表示 已绑定
        //在user里面找是否有m_id
        if (this.btnTxt == "立即解绑") {
          this.untieAccount();
        } else {
          this.bindAccount();
          console.info("立即绑定");
        }
      }
    },
    // 绑定账号
    async bindAccount() {
      console.info("绑定");
      let params = {
        account: this.account,
        password: this.$md5(this.password),
      };
      let { data, message, code } = await bind(params);
      console.info(data);

      this.$router.push({
        path: "/associationSuccess",
        query: {
          name: "bind",
        },
      });

      console.info(data);
    },
    // 解绑账号
    async untieAccount() {
      let params = {
        account: this.account,
        password: this.$md5(this.password),
      };
      let { data, message, code } = await untie(params);
      this.$router.push({
        path: "/associationSuccess",
        query: {
          name: "untie",
        },
      });
    },
    // 微信授权跳转URL
    async getUrl() {
      let params = {
        url: "https://ytl.114family.cn/ytlwap-dev/accountAssociation",
        platform: "",
      };
      let data = await getAuthorizeURL(params);
      window.location.href = data;
    },
    // 微信授权接口
    async getUserByCode() {
      let params = {
        code: this.code,
        platform: "",
      };

      let data = await getUserByCode(params);

      this.getUserData = data;
      let wxToken = data.wxtoken;
      console.info(wxToken);
      sessionStorage.setItem("wxToken", wxToken);
      if (this.getUserData.user.hasOwnProperty("m_id")) {
        this.btnTxt = "立即解绑";
      } else {
        this.btnTxt = "立即绑定";
        console.info("立即绑定");
      }
    },

    // 获取code
    getCode() {
      // 非静默授权，第一次有弹框
      // 截取code
      this.code = this.getUrlParam("code", window.location.href);
      sessionStorage.setItem("code", this.code);
      if (this.code == null || this.code === "") {
        // 如果没有code，则去请求
        this.getUrl();
      } else {
        //如果有code就调微信授权接口
        // alert(this.local_url)
        // window.location.replace(this.local_url);
        this.getUserByCode();
      }
    },
    // 通过固定的参数名截取url中的参数
    getUrlParam(paraName, url) {
      var arrObj = url.split("?");
      if (arrObj.length > 1) {
        var arrPara = arrObj[1].split("&");
        var arr;
        for (var i = 0; i < arrPara.length; i++) {
          arr = arrPara[i].split("=");
          if (arr != null && arr[0] == paraName) {
            return arr[1];
          }
        }
        return "";
      } else {
        return "";
      }
    },
  },
};
