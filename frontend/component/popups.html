    isSHowDialog(city, localArr, lastSetTime) {
      let newtime = new Date().getTime();
      let time = (newtime - lastSetTime) / 1000 / 60 / 60 / 60;
      console.log(newtime);
      console.log(lastSetTime);
      console.log(time);
      if (time >= 24) {
        localArr.forEach((i, index) => {
          if (i.city === city.city) {
            localArr[index].time = new Date().getTime();
          }
        });
        localStorage.setItem("localArr", JSON.stringify(localArr));
        this.popInfo.showPop = true;
      } else {
        this.popInfo.showPop = false;
      }
    },
    async setPopConfig() {
      const res = await await getHomePopup();
      console.log(res);
   //如果返回的是空就表示后台没有配置或没有到时间
      if (JSON.stringify(res) !== "{}") {
        // 当前城市信息
        let provinceCity = JSON.parse(
          localStorage.getItem("114LIFELOCATION")
        ).city;

        // console.log(provinceCity)
        const city = { city: provinceCity };
        const localArr = JSON.parse(localStorage.getItem("localArr")) || [];
        console.log(localArr);
        if (localArr.length > 0) {
          // 是否存在当前城市
          console.log(localArr);
          const isHascity = localArr.filter((i) => i.city === city.city);
          console.log(isHascity);
          if (isHascity.length > 0) {
            // 上次存入时间戳
            const lastSetTime = isHascity[0].time;
            this.isSHowDialog(city, localArr, lastSetTime);
          } else {
            this.popInfo.showPop = true;
            localArr.push({ city: provinceCity, time: new Date().getTime() });
            localStorage.setItem("localArr", JSON.stringify(localArr));
          }
        } else {
          const temp = {
            city: provinceCity,
            time: new Date().getTime(),
          };
          localArr.push(temp);
          localStorage.setItem("localArr", JSON.stringify(localArr));
          // 第一次进入操作
          // alert(1111111111111111111)
          this.popInfo.showPop = true;
        }

        this.popInfo.img = res.pic;
        this.popInfo.url = res.url;
      }
