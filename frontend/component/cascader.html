          parent.children.push(child);

          let c = _.find(children, function (o) {

          parent.children.push(c);

    findChildren(src) {
      let children = [];
      // 全部拿出来
      src.forEach((item) => {
        if (item.children.length) {
          item.children.forEach((child) => {
            children.push(child);
          });
        }
      });
      return children;
    },

    moveUpDown(order, index, i) {
      let item = JSON.stringify(this.tagList[i].circleMenus[index]);

      switch (order) {
        case "up":
          let upItem = JSON.stringify(this.tagList[i].circleMenus[index - 1]);
          this.$set(this.tagList[i].circleMenus, index - 1, JSON.parse(item));
          this.$set(this.tagList[i].circleMenus, index, JSON.parse(upItem));
          break;
        case "down":
          let downItem = JSON.stringify(this.tagList[i].circleMenus[index + 1]);
          this.$set(this.tagList[i].circleMenus, index + 1, JSON.parse(item));
          this.$set(this.tagList[i].circleMenus, index, JSON.parse(downItem));
          break;
        default:
          break;
      }

TIP

单层对象深拷贝 Object.assign(obj1,obj2)
如：{name：1，page：12} 就用Object.assign

多层要么循环获取 要么 JSON.parse(JSON.stringify(val))

splice可以做到每个下标给不同的变量

单选级联选择器：由于多层嵌套 在弹窗里获取的index出错 可以在点击打开弹窗时获取 （this.i）
